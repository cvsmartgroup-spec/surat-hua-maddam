            if (!b) return '0 B';

            if (b < 1048576) return (b / 1024).toFixed(1) + ' KB';
            return (b / 1048576).toFixed(1) + ' MB';
        }

        function showToastApp(msg, err = false) {
            const old = document.querySelector('.toast-app');
            if (old) old.remove();
            const t = document.createElement('div');
            t.className = 'toast-app ' + (err ? 'error' : 'success');
            t.textContent = msg;
            document.body.appendChild(t);
            setTimeout(() => t.remove(), 3000);
        }
